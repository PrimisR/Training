const float C=12.01,H=1.008,O=16.00,N=14.01;

{
	int T;
	int total[100];
	float sum;
	char s[100];
	int len;
	int num;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%s",s);
		len=strlen(s);
		memset(total,0,sizeof(total));
		for(int i=0;i<len;i++)
			if(s[i]>'A'&&s[i]<'Z')
			{
				num=0;
				if((s[i+1]>'A'&&s[i+1]<'Z')||s[i+1]=='\0') total[s[i]]++;
				else
				{
					for(int j=i;s[j+1]>='0'&&s[j+1]<='9';j++)
					{
					num*=10;
					num+=s[j+1]-'0';
					}
					total[s[i]]+=num; 
				}
			} 
		sum=C*total['C']+H*total['H']+O*total['O']+N*total['N'];
		printf("%.3f\n",sum);
	}

#include <stdio.h>

int T;
int n;
int num,bit;
int tot[10];

{
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&n);

		for(int i=1;i<=n;i++)
		{
			num=i;
			while(num)
			{
				bit=num%10;
				tot[bit]++;
				num/=10;
			}//10进制逐位计算
		}
		for(int i=0;i<=9;i++)
		{
			printf("%d",tot[i]);
			if(i!=9)	printf(" ");//格式
		}
		printf("\n");
	}
	return 0;
}

#include <stdio.h>

#include <memory.h> 

char s[100];
char ans[100];
int factor[100];
int len;
int T;
int k;
int ok[100];

{
	scanf("%d",&T);
	while(T--)
	{
		scanf("%s",s);
		getchar();
		len=strlen(s);
		memset(factor,0,sizeof(factor));
		k=0;
		if(len==1)
			printf("1");
		else
		{
		for(int i=len;i>=1;i--)
			if(len%i==0)
				factor[k++]=i;
		memset(ok,1,sizeof(ok));
		for(int i=1;i<k;i++)
		{
			for(int j=0;j<factor[i];j++) ans[j]=s[j];
			for(int j=0;j<len;j++)
				if(ans[j%factor[i]]!=s[j])	ok[i]=0;//取余数周期比较
		}
		for(int i=k-1;i>=0;i--)
		{
			if(ok[i])
			{
				printf("%d",factor[i]);
				break;
			}
		}
		}
		if(T)
			printf("\n\n");
		else
			printf("\n");//输出格式
	}

#include <stdio.h>
#include <string.h>

int s_len;
int t_len;

char s[100000];
char t[100000];

int pos;
int ok;

int main()
{
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	while(scanf("%s %s",s,t)!=EOF)//scanf遇到EOF会返回EOF
	{
		getchar();
		s_len=strlen(s);
		t_len=strlen(t);
		pos=0;
		ok=0;
		for(int i=0;i<s_len;i++)//s[i]
		{
			ok=0;
			for(int j=pos;j<t_len;j++)
			{
				if(s[i]==t[j])
				{
					pos=j+1;//一定小心重复的，小心开始检查的位置
					ok=1;
					break;
				}
			}
			if(!ok)
			{
				printf("No\n");
				break;
			}
		}
		if(ok)
			printf("Yes\n");
	}
	return 0;
}
